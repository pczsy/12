    删除前确认框：

    //找到删除那天记录的按钮，触发点击事件
$(".del").on('click', function () {
        swal({
            title: "操作确认",
            text: "删除后，您将无法恢复此文件！",
            icon: "warning",
            buttons: true,
            dangerMode: true,
        })
            .then((willDelete) => {
                if (willDelete) {
                    $.ajax({
                        url: '/book_del/',
                        type: 'post',
                        //给这个按钮绑定一个删除对象的id，通过this.id获取
                        data: {'book_id': this.id},
                        success:function () {
                            swal("噗！您的虚拟文件已被删除！", {
                            icon: "success",})}
                            })
                        //删除成功后刷新一次页面，显示删除结果
                        .then((value => {
                            location.reload(true)
                        }))
                }
                else {
                    swal("删除未生效！");
                }
            });
    });
