    <!DOCTYPE html>
    <!-- 文档头：告诉浏览器用那种方式编译代码 -->
    <!-- !+tab -->

    <!-- 文档头 -->
    <html>
    <head>
    <meta charset="utf-8">
    <!-- 字符集：具体看公司之前项目设置 -->
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <title>网站标题</title>
    <!-- 网站标题 -->
    <link rel="stylesheet" href="">
    </head>
    <!-- 放置配置文件 -->

    <!-- 放置自己写的代码 -->
    <body>
    <!-- 正确使用标签的目的：为了让浏览器知道每个网页结构的作用 -->


    <!-- 在网页中放置一张图片，img+tab -->
    <!-- 标签属性：k="v" 属性与属性之间用空格隔开，比如k="v" k="v" -->
    <!-- src标签属性：图片的路径。alt：图片不能正常显示时提示的文字。一般有图片的时候，最好加上alt和title -->
    <!-- img标签比较特别，有width和height属性。一般来说，图片的宽和高只设置其中一个，另外一个按照比例自动计算 -->
    <img src="img/1.jpg" alt="这是美女" title="鼠标移入了" width="200">

    <!-- 标题文字，从h1到h6。在一个网页中，h1一般只出现一次，h2最多出现两次 -->
    <h1>这是h1</h1>
    <h2>这是h2</h2>
    <h3>这是h3</h3>

    <!-- 段落文字：在页面中引入一段文字 -->
    <p>这是文字</p>

    <!-- href:要跳转的地址。在新页面中打开，加上target="_blank"，默认是在原来的页面打开 -->
    <a href="https://www.tmall.com/?spm=a220m.1000858.a2226mz.1.d4602a68C2p94H" target="_blank">超链接</a>
    <!-- 阻止链接跳转，加# -->
    <a href="#">超链接2</a>
    </body>
    </html>

    <!--
    网站：有多个网页组成
    网页：图片、文字和超链接组成
    web前端标准：有几项技术组成。html+css+js
    html：超文本标记语言。标记：标签也叫做元素。主要用于页面结构
    css：层叠样式表。主要负责页面的样式
    js：主要负责用户的交互

    -->
