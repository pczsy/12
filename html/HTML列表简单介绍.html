    1.无序列表

    使用标签：<ul>,<li>

    属性：disc，circle，square

    2.有序列表

    使用标签：<ol>，<li>

    属性：A，a，I，i，start

    3.嵌套列表

    使用标签：<ul>，<ol>，<li>

    4.自定义列表

    使用标签：<dl>，<dt>，<dd>

    <!--各个标签使用演示-->

    <!    DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>列表的使用</title>
</head>
<body>
    <ol>
        <li>第一</li>
        <li>第二</li>
        <li>第三</li>
    </ol>
    <br/>
    <!--“<br/>”:换行-->
        <ul>
        <li>第一</li>
        <li>第二</li>
        <li>第三</li>
    </ul>
    <br/>
    <ol type="a">
        <li>
            <ul type="diss">
                <li>第一</li>
                <li>第二</li>
                <li>第三</li>
            </ul>
        </li>
        <li>第二</li>
        <li>第三</li>
    </ol>
    <br/>
    <ol type="I">

            <li>第一</li>
            <li>第二</li>
            <li>第三</li>

    </ol>
    <br/>
    <ol start="3">
        <li>第一</li>
        <li>第二</li>
        <li>第三</li>
    </ol>
    <br/>
    <ul type="diss">
        <li>第一</li>
        <li>第二</li>
        <li>第三</li>
    </ul>
    <br/>
    <ul type="square">
        <li>第一</li>
        <li>第二</li>
        <li>第三</li>
    </ul>
    <br/>
    <ul type="circle">

            <li>第一</li>
            <li>第二</li>
            <li>第三</li>

    </ul>
</body>
</html>
