      background:url(../img/chooseBg.jpg) no-repeat center center;
      background-size:cover;

      height:100%;
      //.listConHeight{
      //   height:9rem;
      //  overflow:auto;
      //}
      .listConHeight {
        position: absolute;
        top: 0;
        left: 0;
        width: 100%;
        height: 100%;
        padding-top: 1.3rem;
        box-sizing: border-box;
        ul {
          width:6.78rem;
          height: 100%;
          margin:0 auto;
          overflow: auto;
          -webkit-overflow-scrolling: touch;
        }
