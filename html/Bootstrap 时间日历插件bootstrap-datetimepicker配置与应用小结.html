    JS代码片段

    时间插件配置

    // 设置开始时间插件

    $('#' + startTimePickerID).datetimepicker({

    language:  'zh-CN',// 默认值: 'en'，设置控件上的文案为中文

    format:'yyyy-mm-dd HH:mm:ss', //格式化，以便精确到秒

    autoclose:true // 选择时间时，点击分后，关闭时间插件框

    });

     

    // 设置结束时间插件

    $('#' + endTimePickerID).datetimepicker({

    language: 'zh-CN',

    format:'yyyy-mm-dd HH:mm:ss',

    autoclose: true // 选择时间时，点击分后，关闭时间插件框}

    });
