        function getUrlData() {
            var url = window.location.search;  //url中?之后的部分
            url = url.substring(1);    //去掉?
            var dataObj = {};
            if (url.indexOf('&') > -1) {
                url = url.split('&');
                for (var i = 0; i < url.length; i++) {
                    var arr = url[i].split('=');
                    dataObj[arr[0]] = arr[1];
                }
            }
            else {
                url = url.split('=');
                dataObj[url[0]] = url[1];
            }
            return dataObj;
        }
