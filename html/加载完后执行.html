 2 　　$(window).load(function(){ 
 3 // 等到整个window页面加载完成后才执行,两者之间的执行顺序是谁在上方谁优先执行
 4 　　　　console.log("jquery===》window load" ); 
 5 　　}) 
 6 　　window.onload = function(){ 
 7 　　　　console.log("页面加载完成====》onload");

 9 
10 //   （比上面的快）    这两种方式是在document加载完成后就执行,两者之间的执行顺序是谁在上方谁优先执行
11 　　$(function(){ 
12 　　　　console.log("jquery====》document onload"); 
13 　　});
